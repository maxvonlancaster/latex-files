\begin{center}
\section*{Операції над множинами.}
\end{center}

\begin{enumerate}



\begin{multicols}{2}
[
\item Довести рівності множин:
]
а) $A\cup (B\cap C) = (A \cup B) \cap (A \cup C)$ \\
в) $A\setminus (B\cap C) = (A \setminus B) \cup (A \setminus C)$ \\
б) $A\cap (B\cup C) = (A \cap B) \cup (A \cap C)$ \\
г) $(A\setminus B)\setminus C = (A\setminus C)\setminus (B\setminus C)$
\end{multicols}



\begin{multicols}{2}
[
\item Довести, використовуючи закон де Моргана:
]
а) $\overline{A\cup (B\cap C)} = (\overline{C} \cup \overline{B}) \cap \overline{A}$ \\
б) $ \overline{(A \cup B) \cap C} = \overline{C }\cup (\overline{A} \cap \overline{B}) $
\end{multicols}


\begin{multicols}{2}
[
\item Що можна сказати про множини $A$ та $B$, якщо:
]
а) $A \bigtriangleup B = A$ \\
в) $A \cup B = A \cap B$ \\
б) $A \bigtriangleup B = \overline{A}$ \\
г) $A \setminus B = B \setminus A$
\end{multicols}


\begin{multicols}{2}
[
\item Використовуючи закон дистрибутивності (див. 1) довести:
]
а) $(A\cup B)\cap (A\cup \overline{B})$ \\
в) $(A\cap B \cap C)\cup (\overline{A}\cap B \cap C)\cup \overline{B} \cup \overline{C} = U$ \\
б) $(\overline{A}\cup B)\cap A = A\cap B$
\end{multicols}


\end{enumerate}
