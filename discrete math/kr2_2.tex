\section*{Модульна контрольна робота №1. Варіант 1.}
\end{center}

\begin{enumerate}

\item Дано множини $A=\{1,3,7,137\}$, $B=\{3,7,23\}$, та $C=\{0,7,23,2020\}$. Знайдіть $$(A\cap B)\cup C$$


\item Доведіть рівність, використовуючи логічні таблиці: $$A\cup (A\cap B) = A.$$


\item Наведіть приклад рефлексивного відношення на довільній множині. Покажіть, що воно є рефлексивним.





\item Наведіть приклад ґратки.

\end{enumerate}

\begin{center}
\section*{Модульна контрольна робота №1. Варіант 2.}
\end{center}

\begin{enumerate}

\item Дано множини $A=\{1,3,7,137\}$, $B=\{3,7,23\}$ та $C=\{0,1,3,23\}$ . Знайдіть $$A\setminus (B\setminus C ).$$

\item Доведіть рівність, використовуючи логічні таблиці: $$A\cap (A\cup B) = A.$$

\item Наведіть приклад симетричного відношення на довільній множині. Покажіть, що воно є симетричним.




\item Наведіть приклад ланцюга в частково впорядкованій множині.

\end{enumerate}
