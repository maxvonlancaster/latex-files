\section*{Біном Ньютона. Комбінаторні рівняння.}
\end{center}

\begin{enumerate}


\item Знайти розклад бінома:\\
а) $(2x-4)^{4}$; б) $(\frac{a}{2}+2b)^{5}$; в) $(2-x^{4})^{5}$; г) $(a+2b)^{7}$.

\item Знайти коефіцієнт при $x^{3}$ у многочлені:\\
а) $x(2-3x)^{5}+x^{3}(1+2x^{2})^{7}-x^{2}(5+3x^{3})^{4}$; \\
б) $(5+2x)^{16}$

\item Обмежившись двома членами у розкладі бінома, наближено обчислити: \\
а) $(2,003)^{10}$; б) $(4,98)^{4}$; в) $(1,0018)^{5}$.

\item Розв'язати рівняння відносно натурального $m$:\\
а) $C_{m+1}^{m-1}+C_{m}^{m-2}=9m+10$\\
б) $C_{m}^{3}+C_{m}^{4}=11C_{m+1}^{2}$.

\item Знайти коефіцієнт при $x^{k}$ у розкладі полінома:\\
а) $(1+x^{2}+x^{3})^{7}$, $k=11$; б) $(1+2x^{2}-3x^{4})^{10}$, $k=8$.

\end{enumerate}
