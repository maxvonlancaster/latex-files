\section*{Варіант 1.}
\end{center}

\begin{enumerate}
\item Дано множини $A=\{1,3,7,137\}$, $B=\{3,7,23\}$, $A=\{0,1,3,23\}$ та $A=\{0,7,23,2020\}$. Знайти: 
    
    а) $(A\cap B)\cup D$
    
    б) $A\setminus (B\setminus(C\setminus D))$

\item Запишіть всі підмножини множини $\{\{1,2\},3\}$.

\item Скільки існує перестановок чисел $1,2,...,9$, у яких цифра 2 займає друге місце, а цифра 5~--- п'яте або шосте.

\item Знайдіть розклад бінома: $$(\frac{a}{2}+2b)^{5}.$$

\item Знайдіть коефіцієнт при $x^{3}$ у многочлені $$x(2-3x)^{5}+x^{3}(1+2x^{2})^{7}-x^{2}(5+3x^{3})^{4}.$$
\end{enumerate}

\section*{Варіант 2.}
\end{center}

\begin{enumerate}
\item Дано множини $A=\{1,3,7,137\}$, $B=\{3,7,23\}$, $A=\{0,1,3,23\}$ та $A=\{0,7,23,2020\}$. Знайти:

    а) $C\cap(D\cap B)$

    б) $(A\cup B)\setminus (C\cap D)$

\item Наведіть приклад множини, що спростовує рівність $(A\cup B)\setminus B = A$

\item Скількома способами можна із 6 інженерів та 15 лаборантів зібрати бригаду, яка складалася б із 2 інженерів та 4 лаборантів.

\item Знайдіть розклад бінома: $$(2-x^{4})^{5}.$$

\item Знайдіть коефіцієнт при $x^{3}$ у многочлені $$x(2-3x)^{5}+x^{3}(1+2x^{2})^{7}-x^{2}(5+3x^{3})^{4}.$$
\end{enumerate}
