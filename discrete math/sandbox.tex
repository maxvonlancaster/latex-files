\begin{center}
\textbf{Теорія множин}
\end{center}


\begin{enumerate}
\item Дано множини $A=\{1,3,7,137\}$, $B=\{3,7,23\}$, $A=\{0,1,3,23\}$ та $A=\{0,7,23,2020\}$. Знайти:\\
    а) $(A\cap B)\cup D$;    б) $A\setminus (B\setminus(C\setminus D))$

\item Запишіть всі підмножини множини а) $\{\{1,2\},3\}$.

\item Довести рівності множин, використовуючи таблиці істинності: \\
а) $A\cup (B\cap C) = (A \cup B) \cap (A \cup C)$ ;
б) $A\setminus (B\cap C) = (A \setminus B) \cup (A \setminus C)$; \\
в) $A\cap (B\cup C) = (A \cap B) \cup (A \cap C)$ ;
г) $(A\setminus B)\setminus C = (A\setminus C)\setminus (B\setminus C)$.

\item Довести рівності множин, використовуючи еквівалентні перетворення: \\
а) $A \setminus (A \setminus (A \ B)) = A \setminus B$;
б) $ (A\setminus B)\setminus (\overline{A}\setminus B)= A\setminus B$; \\
в) $A\triangle (A\setminus B)=A\cap B$;
г) $(A\cap B)\triangle (B\setminus A)=B$;\\
ґ) $ (A\triangle B)\setminus A = B\setminus A$;

\item Довести, використовуючи закон де Моргана:\\
а) $\overline{A\cup (B\cap C)} = (\overline{C} \cup \overline{B}) \cap \overline{A}$;
б) $ \overline{(A \cup B) \cap C} = \overline{C }\cup (\overline{A} \cap \overline{B})$.

\item Нехай дано множину $A=\{a,b,c,d,e\}$, а \\ $R_{1}=\{(a,b),(a,c),(b,b),(c,c),(e,e),(d,d),(d,a),(d,b),(c,d),(e,d)\}$, \\
    $R_{2}=\{(a,d),(d,a),(c,d),(c,c),(b,b)\}$~--- бінарні відношення на $A$\\
    Побудувати: а) $\overline{R_{1}}$  б) $R_{1}\cap R_{2}$  в) $R_{1}\cup R_{2}$  г) $R_{1}\setminus R_{2}$
    ґ) $R_{1}\bigtriangleup R_{2}$ \\ д) Яке з відношень з попередніх пунктів буде рефлексивним, яке симетричним, а яке транзитивним? \\
    е) Побудувати матрицю та граф відношення $R_{1}$, $R_{2}$, $R_{1}\cup R_{2}$, $R_{1}\cap R_{2}$.\\
    є) Побудувати матрицю та граф відношення $R_{1}^{2}$, $R_{1}^{3}$.


\item Дати геометричну інтерпретацію відношень, та навести їхні властивості (рефлексивність, симетричність, транзитивність): \\
а) $\{(x,y)\in \mathbb{R}^{2}| y=x \}$;  б) $\{(x,y)\in \mathbb{R}^{2}| y>x \}$;
в) $\{(x,y)\in \mathbb{R}^{2}| 0\leq x\leq 1\textmd{ або }0\leq y\leq 1\}$;  \\
г) $\{(x,y)\in \mathbb{R}^{2}| x^{2}+y^{2}=1\}$;  ґ) $\{(x,y)\in \mathbb{R}^{2}| y^{2}=x^{2} \}$;

\item Навести приклад двох антирефлексивних відображень на довільній множині, композиція яких не буде антирефлексивним відображенням.

\item Довести, що наступне відношення $R$ на $\mathbb{N} \times \mathbb{N}$ є еквівалентним: \\
    а) $((x_{1},y_{1}),(x_{2},y_{2}))\in R \Longleftrightarrow x_{1}\cdot y_{2}=y_{1}\cdot x_{2}$;\\
    б) $((x_{1},y_{1}),(x_{2},y_{2}))\in R \Longleftrightarrow x_{1}+ y_{1}=x_{2}+ y_{2}$

\item Нехай дано множину $A=\{a,b\}$, побудуйте на ній всі бінарні відношення, та вкажіть, які серед них будуть: \\
    а) рефлексивними; б) антирефлексивними; в) симетричними; г) антисиметричними; \\ ґ) транзитивними;  д) еквівалентними; е) часткового порядку;

\item Чи є наступні відношення еквівалентними? Відношеннями часткового порядку? \\
а) відношення $n\vdots m$ ($m$ ділить націло $n$) на множині натуральних чисел; \\
б) відношення на множині натуральних чисел $mRn \Longleftrightarrow \frac{m}{n}=2^{k}$, $k\in \mathbb{Z}$.

\item Нехай $R$~--- бінарне відношення для дійсних чисел, таке, що $$aRb \Longleftrightarrow \frac{a}{a^{2}+1}\leq \frac{b}{b^{2}+1}.$$
Чи буде $R$ відношенням часткового порядку на інтервалах:\\
    а) $(1,\infty)$;  б) $[1,\infty)$;  в) $(-1,1)$;  г) $[-1,1]$;  ґ) $(-\infty,1]$;  д) $(-\infty,0)$.

\item Побудуйте діаграми Гассе для наступних частково впорядкованих множин: \\
а) $(\mathcal{P}(A),\subseteq)$, де $\mathcal{P}(A)$~--- булеан множини $A$ (всі підмножини множини $A$), $A=\{a,b,c\}$, $\subseteq$~---
відношення включення ("є підмножиною") для множин;\\
б) $(A,|)$, де $A$~--- множина всіх дільників числа $18$, а $|$~--- відношення подільності чисел ($a|b$ <=> $a$ ділить $b$);\\
в) $(A,|)$, де $A$~--- множина всіх дільників числа $60$, а $|$~--- відношення подільності чисел ($a|b$ <=> $a$ ділить $b$);\\
г) $(D,\leq)$, де $D=\{(1,1),(9,1),(3,2),(1,3),(10,1),(4,6),(7,5),(0,7),(8,8)\}$~--- множина пар дійсних чисел, і
$(x_{1},y_{1})\leq (x_{2},y_{2}) <=> x_{1}\leq x_{2}\textmd{ і }y_{1}\leq y_{2}$;\\
Також:\\
ґ) Вкажіть висоти та ширини ЧВМ а)-г);\\
д) Вкажіть всі ланцюги та антиланцюги в ЧВМ а)-г);\\
е) Вкажіть, які з ЧВМ а)-г) будуть ґратками.\\


\end{enumerate}
