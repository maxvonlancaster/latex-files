\section*{Поліноми.}
\end{center}


\begin{enumerate}
\item Розкладіть на множники $2x^{3}+x^{2}+x-1$.

\item Розкладіть на множники $a^{10}+a^{5}+1$.

\item Розв'яжіть рівняння $(1+x+x^{2})(1+x+...+x^{10})=(1+x+...+x^{6})^{2}$

\item Розкладіть на множники $(a+b+c)^{3}-a^{3}-b^{3}-c^{3}$.

\item Многочлен $p(x)$ дає остачу $2$ при діленні на $x-1$ і остау $1$ при діленні на $x-2$. Яка остача $p(x)$ при діленні на $(x-1)(x-2)$.

\end{enumerate}
