\section*{Тригонометрія.}
\end{center}


\begin{enumerate}
\item Обчисліть $\cos20^{\circ} \cos40^{\circ} \cos60^{\circ} \cos80^{\circ}$

\item Розв'яжіть рівняння $\sin^{4}x+\cos^{4}x=a$

\item Доведіть рівність $$\sin \alpha + \sin \beta + \sin \gamma - \sin (\alpha + \beta + \gamma) = 4\sin \frac{\alpha+\beta}{2} \sin \frac{\gamma+\beta}{2} \sin \frac{\alpha+\gamma}{2}$$
    
    
\item Нехай $\sin 18^{\circ}=x$. Доведіть, що $4x^{2}+2x=1$.
    
\item Нехай $\alpha+\beta+\gamma = \pi$. Доведіть рівність $$\sin \alpha + \sin \beta + \sin \gamma = 4\sin \frac{\alpha}{2} \sin \frac{\beta}{2} \sin \frac{\gamma}{2}$$

\item Обчисліть $\cos\frac{2\pi}{7} + \cos\frac{4\pi}{7} + \cos\frac{6\pi}{7}$



\end{enumerate}
