\begin{frame}{Поліном Бернштейна.}
\begin{block}{}
Поліномом Бернштейна $n$~--го порядку для неперервної функції $f(x)\in  СC([0,1])$
\end{block}

\begin{equation}
B_{n}(x)=\sum\limits_{k=0}^{n} f(\frac{k}{n}) C_{n}^{k} x^{k} (1-x)^{n-k}.
\end{equation}

\end{frame}



\begin{frame}{Логістичне відображення.}

\begin{block}{}
Логістичне відображення при натуральному $i$, $r\in (0,4)$.
\end{block}

\begin{equation}
x_{i+1}=rx_{i}(1-x_{i}).
\end{equation}

\end{frame}



\begin{frame}{Цікавий розвиток.}

\begin{equation}
x_{i+1}(x)=\sum\limits_{k=0}^{n} f(\frac{k}{n}) C_{n}^{k} x_{i}^{k} (1-x_{i})^{n-k}. 
\end{equation}

\end{frame}
