\section*{Функції однієї змінної. Функціональні рівняння.}
\end{center}

\medskip

\hrule height 1pt
\vskip 3pt \hrule

\medskip

\textit{\textbf{Теорія.}} \textit{Функціональним рівнянням} називаются рівняння, в яких
невідома функція пов’язана з відомими з допомогою арифметичних дій та
операції утворення складеної функції.

Найпоширенішими способами розв'язання функціональних рівнянь є \textit{спосіб підстановок} (підставляємо замість $x$ певний вираз, що залежить від $t$) та \textit{спосіб невизначених коефіцієнтів} (підставляємо замість $f(x)$ функцію вигляду $f(x)=ax+b$ чи $f(x)=ax^{2}+bx+c$ та розв'язуємо відносно коефіцієнтів)





\medskip
\medskip

\hrule height 1pt
\vskip 3pt \hrule

\medskip
\medskip
\medskip


\begin{enumerate}
\item Нехай для функції $f$ та довільних чисел $x$ та $y$ виконується $f(xy)=f(x)+f(y).$ Знайти $f(100)$, якщо $f(\frac{1}{100})=1$.

\item Знайдіть всі функції $f(x)$ визначені для всіх дійсних $x$, такі, що виконується рівність $2f(x)+f(1-x)=x^{2}.$

\item Знайти всі неперервні функції, що задовільняють $f(f(x))=f(x)+x$


\item Знайдіть усі многочлени $f(x)$, для яких $f(x-y)=f(x)+f(y) + 4y(x+1)$


\item Знайдіть всі такі функції $f(x)$, що для довільного $x$ виконується: $f(2x+1)=4x^{2}+14x+7.$

\item Нехай для функції $f$ та довільного $x$ виконується $f(x)+(x+\frac{1}{2})f(1-x)=1.$

а) Знайти $f(0)$ і $f(1)$.

б) Знайти $f(x)$.

\item Нехай для функції $F$ та довільного $x$ виконується $F(x+1)F(x)+F(x+1)+1=0.$ Довести, що $F$ не може бути неперервною.

\item Знайти всі функції $f$, такі, що для довільних $x,y,z$ виконується нерівність $f(x+y)+f(y+z)+f(z+y)\geq 3f(x+2y+3z).$

\item Для яких дійсних $\alpha$ існує функція $f$ не рівна константі, така, що для довільних $x,y$ виконується $f(\alpha(x+y))=f(x)+f(y).$

\item Нехай функція $f(x)$ визначена для довільного $x$ та при $x\neq 1$ виконується $$(x-1)f(\frac{x+1}{x-1})-f(x)=x.$$ Знайдіть а) $f(-1)$; б) $f(x)$.

\item Нехай функція $f(x)$ задана для всіх дійсних чисел, та для довільних $x,y$ при $x>y$ виконується $(f(x))^{2}\leq f(y).$ Доведіть,  що множина всіх значень функції $f$ належить відрізку $[0,1]$

\end{enumerate}
