\begin{center}
\section*{Нерівності}
\end{center}

\medskip
\hrule
\medskip
\medskip
\bigskip

 \textbf{\textit{Нерівність Коші:}} $x^{2}+y^{2}\geq 2xy.$

 \textbf{\textit{Нерівність Бернуллі:}} $(1+x)^{\alpha}\leq 1+\alpha x$, $x\geq -1$, $0<\alpha <1$.

 \textbf{\textit{Нерівність Коші-Буняковського-Шварца:}} $$(a_{1}^{2}+a_{2}^{2}+...+a_{n}^{2})(b_{1}^{2}+b_{2}^{2}+...+b_{n}^{2}) \geq (a_{1}b_{1}+a_{2}b_{2}+...+a_{n}b_{n})^{2}.$$

 \textbf{\textit{Нерівність перестановок:}} Нехай $x_{1}\geq x_{2}\geq ... \geq x_{n}$ та $y_{1}\geq y_{2}\geq ... \geq y_{n}$. Тоді для кожної перестановки $\sigma = \{\sigma_{1},\sigma_{2},...,\sigma_{n}\}$ чисел $(1,2,...,n)$ виконується: $$x_{1}y_{1}+x_{2}y_{2}+...+x_{n}y_{n}\geq x_{1}y_{\sigma_{1}}+x_{1}y_{\sigma_{2}}+...+x_{1}y_{\sigma_{n}}\geq x_{1}y_{n}+x_{2}y_{n-1}+...+x_{n}y_{1}.$$

\textbf{\textit{Нерівність Шура:}} $x^{t}(x-y)(x-z) + y^{t}(y-x)(y-z) + z^{t}(z-y)(z-x) \geq 0$.

\bigskip
\bigskip

\medskip


\begin{enumerate}
\item Нехай $a,b,c$~--- додатні числа, сума яких рівна 1. Довести, що $$(1+a)(1+b)(1+c)\geq 8(1-a)(1-b)(1-c)$$

\item Доведіть для додатніх $x,y,z$: $$x(x - z)^{
2} + y(y - z)^{
2} \geq (x - z)(y - z)(x + y - z).$$

\item Доведіть для $x,y,z$ таких, що $x+y+z$, наступну нерівність: $$6(x^{3}+y^{3}+z^{3})+1\geq 5(x^{2}+y^{2}+z^{2})$$

\item Для $a,b,c > 0$ доведіть: $$(a+b)(b+c)(c+a) \geq 8abc$$
    
\item Доведіть, що для любих $x,y,z$ справджується нерівність $$x^{4}+y^{4}+z^{4}\geq xyz(x+y+z)$$.
    
\item Нехай $a^{5}-a^{3}+a=2$. Довести, що $a^{6}>3$

\item Доведіть, що $x^{4}+y^{4}+8\geq 8xy$ для любих $x$ та $y$.
    
\item $x,y,z\geq 0$. Доведіть, що $xy+yz+zx\geq x\sqrt{yz}+y\sqrt{zx}+z\sqrt{xy}$.

\item Для довільних $x,y,z>0$ таких, що $xyz=1$ доведіть: $$\frac{x^{3}}{(1+y)(1+z)} + \frac{y^{3}}{(1+x)(1+z)} + \frac{z^{3}}{(1+y)(1+x)}\geq \frac{3}{4}$$
\end{enumerate}
