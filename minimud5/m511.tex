\section*{Математична індукція.}
\end{center}

\medskip

\hrule height 1pt
\vskip 3pt \hrule

\medskip
\medskip

\textit{\textbf{Математична індукція} ~--- Твердження $A(n)$ , залежне від натурального параметра $n$ , вважається доведеним для всіх натуральних $n$ , якщо воно доведене для $n =1$ і з припущення про виконання твердження $A(k)$ для якого-небудь натурального $k$ виведено, що справджується твердження $A(k+1)$ для натурального числа $k+1$.\\}




\medskip
\medskip
\hrule
\medskip
\medskip


\begin{enumerate}
\item Доведіть, що при всіх натуральних $n$ число  $5^{n}+ 2\cdot 3^{n-1}+ 1$ ділиться на $8$.

\item Доведіть, що $1^{2}+ 2^{2}+ … + n^{2}= \frac{n (n + 1)(2n + 1)}{6}$.

\item Доведіть, що  $1\cdot 2\cdot 3 + 2\cdot 3\cdot 4 + … n (n + 1)(n + 2) = \frac{n (n + 1)(n + 2)(n + 3)}{4}.$


\item Доведіть, що $$\frac{1}{1\cdot 2\cdot3} + \frac{1}{2\cdot 3\cdot4}+...+\frac{1}{n(n+1)(n+2)} = \frac{n(n+3)}{4(n+1)(n+2)}.$$

\item Обчисліть суму: $1^{3}+ 2^{3}+ … + n^{3}.$

\item Послідовність $\{a_{n}\}$ така, що $a_{1} = 3$, $a_{2} = 5$, $a_{n+2} = 3a_{n+1}- 2a_{n}$. Доведіть, що $a_{n} = 2^{n}+ 1$ для всіх натуральних $n$.

\item Число $x+\frac{1}{x}$ є цілим. Доведіть, що при кожному натуральному $n$ число $x^{n}+\frac{1}{x^{n}}$~--- також ціле.

\item Числа $N_{1}$, $N_{2}$, … $N_{n}$ є сумами квадратів двох цілих чисел. Доведіть, що і їх добуток є сумою квадратів двох цілих чисел.


\item Доведіть, що при всіх натуральних $n \geq 2$ виконується нерівність $2^{n}>1+n\sqrt{2^{n-1}}$.
    
\item Довести, що $2^{n+m-2}\geq mn$ для будь-яких натуральних $m$ та $n$.

\end{enumerate}
