\begin{frame}\frametitle{Позначення}

\begin{exampleblock}{}

-- $C(X,Y)$ -- простір неперервних відображень $f: X\rightarrow Y$;

-- $B_1(X,Y)$ -- простір відображень першого класу Бера, тобто поточкових послідовностей границь послідовностей неперервних функцій $f_n: X\rightarrow Y$;

-- $f: X\times Y\rightarrow Z$, $f(x,y)=f^x(y)=f_y(x)$;

-- $CC(X\times Y, Z)=$$$=\{f\in Z^{X\times Y}: (\forall x\in X) (f^x \in C(Y, Z))\wedge (\forall y\in Y)(f_y\in C(X, Z)) \}$$ -- cукупність усіх нарізно неперервних відображень $f: X\times Y\rightarrow Z$;

-- $g(x)=f(x,x)$ -- діагональ відображення $f: X^2 \rightarrow Z$.

\end{exampleblock}

\begin{frame}\frametitle{Результати Бера і Лебеґа та їх розвиток}

\begin{alertblock}{\centerline{Теорема 1. (Р.Бер, 1898, 1899)}}

Функція $g: [a,b] \rightarrow \mathbb{R}$ є діагоналлю деякої нарізно неперервної функції $f: [a,b]^2 \rightarrow \mathbb{R}$ тоді і тільки тоді, коли вона є першого класу Бера.

\end{alertblock}

\begin{alertblock}{\centerline{Теорема 2. (A.Лебеґ, 1898)}}

$CC([a,b]\times [c,d], \mathbb{R})\subseteq B_1 ([a,b]\times [c,d], \mathbb{R})$.

\end{alertblock}

\begin{block}{\centerline{Розвиток}}

Г.Ган, К.Куратовський, Д.Монтґомері, В.Моран, Б.Джонсон, Ж.Сан-Ремо, В.Рудін, Ґ.Вера, Р.Генсел, В.Маслюченко, О.Собчук, О.Карлова, Т.Банах та ін.

\end{block}

\end{frame}


\begin{frame}\frametitle{Узагальнення результату Гана}

\begin{block}{\,}

8. Hahn H. Reele Funktionen. 1. Teil. Punktfunktionen. -- Leipzig: Academische Verlagsgesellschaft M.B.H., 1932. -- 416S.

9. Глушко Г.А., Маслюченко В.К. Функції Гана і класифікація Бера// Мат.Студії. --2011. -- 36, №1. -- С.97-106.

\end{block}


\begin{alertblock}{\, }

 $(X,d)$ -- метричний простір,  $x'x''=d(x', x'')$,  $a_1,...,a_n$ -- різні точки з $X$,
$$
\psi(x)=\min\{xa_1,...,xa_n\},\,\,\,\psi_k(x)=\max\{2\psi(x)-xa_k,0\},
$$

$$
\,\,\,\varphi_{n,k}(x)=\frac{\psi_k(x)}{\sum\limits_{i=1}^n \psi_i(x)}\,\,\,{\mbox{при}}\,\,\,x\neq a_k\,\,\,{\rm i}\,\,\, \varphi_{n,k}(a_k)=1.
$$

$Z$ -- топологічний векторний простір, $g\in C(X,Z)$,

$$g_n(x)=H_ng(x)=\sum\limits_{k=1}^{n}\varphi_{n,k}(x)g(a_k),$$
$H_n: C(X,Z)\rightarrow C(X,Z)$ -- оператор Гана.
\end{alertblock}

\end{frame}

\begin{frame}\frametitle{Узагальнення результату Гана}

\begin{alertblock}{\centerline{Теорема 3}}

Нехай $X$ -- сепарабельний метричний простір, $Z$ -- локально опуклий простір над полем $\mathbb{K}=\mathbb{R}$ або $\mathbb{C}$ i $A=\{a_k:k\in \mathbb{N}\}$ -- всюди щільна в $X$ множина, що
 складається з різних точок $a_k$. За кожним набором точок $a_1,...,a_n$ побудуємо функції $\varphi_{n,k}$ і відповідний оператор Гана $H_n$.

 \qquad a). Нехай $g \in C(X,Z)$ i $g_n=H_ng$. Тоді $g_n\in C(X,Z)$ і для кожного $x\in X$ $$g_n(x)\rightarrow g(x)$$
 при $n\rightarrow\infty$ у просторі $Z$.

 \qquad б). Нехай $Y$ -- топологічний простір, $f\in CC(X\times Y,Z)$ i
 $$f_n(x,y)=(H_nf_y)(x)$$
 для довільних $(x,y)\in X\times Y$. Тоді $f_n\in C(X\times Y,Z)$ i
 $f_n(p)\rightarrow f(p)$
 у просторі $Z$ для кожної точки $p\in X\times Y$. Зокрема, $f\in B_1(X\times Y,Z)$ i $CC(X\times Y,Z)\subseteq B_1(X\times Y,Z)$.

\end{alertblock}


\end{frame}

\begin{frame}\frametitle{Нова редакція результату Рудіна}

\begin{block}{\,}

10. Rudin W. Lebesgue first theorem// Math. Analysis and Aplications, Part B. Edited by Nachbin. Adv. in Math. Supplem. Studies 78. - Academic Press, 1981. - P.741-747.

\end{block}

\begin{alertblock}{\, }

Нехай $X$ -- метричний простір. Для кожного номера $n$ розглянемо покриття $\mathcal{B}_n$ простору $X$ відкритими кулями
$$B(x, \frac{1}{n})=\{u\in X: ux<\frac{1}{n}\},$$
де $x\in X$. Згідно з теоремою Стоуна у кожне покриття $\mathcal{B}_n$ можна вписати відкрите локально скінченне покриття $\mathcal{U}_n$. За теоремою Майкла для кожного $n$ існує розбиття одиниці $(\varphi_{n,i})_{i\in I_n}$, яке підпорядковане покриттю $\mathcal{U}_n$ та складається з ненульових неперервних функцій $\varphi_{n,i}: X\rightarrow [0,1]$. Виберемо в кожному з носіїв $supp \varphi_{n,i}=\{x\in X: \varphi_{n,i}(x)\neq 0\}$ деяку точку $x_{n,i}$.

\end{alertblock}

\end{frame}

\begin{frame}\frametitle{Нова редакція результату Рудіна}

\begin{block}{\, }

Нехай $Z$ -- топологічний векторний простір і $g: X\rightarrow Y$ -- неперервне відображення. Для довільного номера $n$ і для кожного $x$ з $X$ покладемо

$$g_n(x)=(R_ng)(x)=\sum\limits_{i\in I_{n}}\varphi_{n,i}(x)g(x_{n,i}).$$

$R_n: C(X,Z)\rightarrow C(X,Z)$ -- оператори Рудіна.

\end{block}

\begin{alertblock}{\centerline{Теорема 4 }}

Нехай $X$ -- метричний простір, $Z$ -- локально опуклий простір і $R_n$ -- оператори Рудіна.

\qquad a). Нехай $g \in C(X,Z)$ i $g_n=R_ng$. Тоді $g_n(x)\rightarrow g(x)$ при $n\rightarrow\infty$ для кожного $x$ з $X$.

\qquad б). Нехай $Y$ -- топологічний простір, $f\in CC (X\times Y, Z)$ і $f_n(x,y)=(R_nf_y)(x)$. Тоді  $f_n\in C(X\times Y, Z)$ i $f_n(x,y)\rightarrow f(x,y)$ на $X\times Y$, зокрема $CC(X\times Y, Z)\subseteq B_1(X\times Y, Z)$.

\end{alertblock}



\end{frame}

\begin{frame}\frametitle{$\alpha\beta$-неперервні оператори та $\alpha\beta$-простори}

\begin{frame}\frametitle{Зв'язок між $s$-просторами та $as$-просторами}

\begin{alertblock}{\centerline{}} % ось як в порожніх дужках
Нехай $L$ -- всюди щільний лінійний підпростір простору $C_u(Y)$, $s=\alpha\beta$. Підпростір $L$  називається

--  {$s$-простором} для топологічного простору $X$, якщо для кожного $\alpha$-неперервного відображення $\varphi: X\rightarrow C(Y)$ існує послідовність $\beta$-неперевних відображень  $\varphi_n: X\rightarrow L$, така, що $\varphi_n (x)\rightarrow \varphi(x)$ в $C_u(Y)$ на $X$;

-- {$as$-простором}, якщо існує така послідовність $s$-неперервних операторів $A_n: C(Y)\rightarrow L$, що $A_ng\rightarrow g$ в $C_u(Y)$ для кожного $g\in C(Y)$.
\end{alertblock}

\begin{exampleblock}{\centerline{Теорема 7}}
Нехай  $Y$ -- компактний простір і $L$ -- підпростір простору $C(Y)$ i $s\in~\{pp,pu,up,uu\}$. Тоді наступні властивості рівносильні:

(і) $L$ є $as$-простором;

(іі) $L$ є $s$-простором для кожного топологічного простору $X$.
\end{exampleblock}

\end{frame}


\begin{frame}\frametitle{Наближення тригонометричними поліномами}

\begin{block}

14. Волошин Г.А., Маслюченко В.К. Про наближення нарізно неперервних функцій, $2\pi$-періодичних відносно другої змінної//Карп. матем.публ. -- 2010. -- {\bf{2}}, №1 -- С. 4-14.

\end{block}

\begin{exampleblock}{\centerline{}}

-- $C_{2\pi}$  --  банаховий простір всіх неперервних $2\pi$-періодичних функцій $g: \mathbb{R}\rightarrow \mathbb{R}$ з нормою $\|g\|=\max_{y\in \mathbb{R}}|g(y)|=\max_{0\leq y\leq 2\pi}|g(y)|;$

-- $CC_{2\pi}(X \times \mathbb{R})$ -- сукупність всіх відображень $f:X \times \mathbb{R} \to \mathbb{R}$, які
неперервні відносно першої змінної і неперервними та $2\pi$-періодичними   відносно другої;

-- $T$ -- сукупність усіх тригонометричних поліномів $g: \mathbb{R}\rightarrow \mathbb{R}$, тобто функцій
$g(y)=~a_0+\sum\limits_{k=1}^{n}(a_k cosky + b_k sin ky )$;

-- $CT(X \times \mathbb{R})$ -- це множина тих функцій $f: X \times \mathbb{R}\rightarrow \mathbb{R}$, для яких функції $f_y:  \mathbb{X}\rightarrow \mathbb{R}$  неперервні для кожного $y\in \mathbb{R}$, а функції $f^x: \mathbb{R}\rightarrow \mathbb{R}$ -- це тригонометричнi поліноми для кожного $x\in X$.

\end{exampleblock}

\begin{alertblock}

К.Вейєрштрасс: $g\in C_{2\pi}\Rightarrow \exists( g_n)_{n=1}^{\infty}| g_n\in T\, i \, g_n\rightrightarrows g \, \mbox{на} \, \mathbb{R}$.

\end{alertblock}

\begin{frame}\frametitle{Застосування многочленів Фейєра }

\begin{alertblock}

Многочлени Фейєра
$(F_n g)(x)=\frac{1}{\pi}\int \limits_{-\pi}^\pi g(x+t)K_n(t)dt,$
де $K_n(t)=\frac{2}{n+1}\left(\frac{sin (n+1)\frac{t}{2}}{2 sin \frac{t}{2}}\right)^2$ -- ядро Фейєра.
\end{alertblock}


\begin{exampleblock}{  \centerline{Теорема 8}} % ось як в порожніх дужках

а). Оператори Фейєра $F_n$ є $uu$-неперервними, але не є $pp$-неперервними.

б). Для довільного топологічного простору $X$ і кожної сукупно неперервної і ${2\pi}$-періодичної відносно  другої змінної функції $f: X \times \mathbb{R} \rightarrow \mathbb{R}$ функції
$$f_n(x,y)=(F_n f^x)(y)$$
сукупно неперервні і належать до простору $CT(X\times \mathbb{R})$, причому $f^x_n\rightrightarrows f^x$ на $\mathbb{R}$ для кожного $x\in X$.
\end{exampleblock}

\end{frame}

\begin{frame}{Лінійна інтерполяція}
\begin{exampleblock}{}
Нехая $A$~--- скінченна підмножина скінченного інтервалу $(a,b)$ числової прямої і $\widetilde{A}=A\cup \{a,b\}$. Занумеруємо елементи $\widetilde{A}$ у порядку зростання: $\widetilde{A}=\{a_{k}: k=0,1,...,n+1\}$. Тоді $A=\{a_{k}:k=1,...,n\}.$

Кожній функції $g:[a,b]\rightarrow \mathbb{R}$ поставимо у відповідність функцію $h=L_{A}g$, графіком якої є ламана з вузлами $(a_{k}, g(a_{k}))$.
\end{exampleblock}

\begin{exampleblock}{  \centerline{Лема }}
Нехай $A=\{a_{k}:k\in \mathbb{N}\}$~--- зліченна підмножина інтервала $(a,b)$, яка всюдищільна в ньому, яка всюди щільна в ньому, причому $a_{k}\neq a_{j}$, при $k\neq j$, $A_{n}=\{a_{1},...,a_{n}\}$, $g\in C[a,b]$ і $g_{n}=L_{A_{n}}g$. Тоді $g_{n}\rightrightarrows g$ на $[a,b]$.
\end{exampleblock}



\end{frame}

\begin{frame}\frametitle{Функціонально обернена задача для сукупно неперервних функцій}
\begin{exampleblock}{}
Якщо для простоти розглянути функцію $f:[0,1]^{2}\rightarrow \mathbb{R}$, яка неперервна відносно другої змінної, то для неї можна визначити функції $\alpha_{n}(x)= E_{n}(f^{x})$ на $[0,1]$, де  $E_{n}$ -- найкраще рівномірне наближення многочленами степеня $\leq n$, а  $f^{x}(y)=f(x,y)$ на $[0,1]$. Можна показати, що для сукупно неперервної функції $f$ функцій $\alpha_{n}:[0,1]^{2}\rightarrow \mathbb{R}$ неперервні. Аналогічно до оберненої задачі Бернштейна можна поставити питання [13]: чи для кожної спадної послідовності невід'ємних неперервних функцій  $\alpha_{n}:[0,1]^{2}\rightarrow \mathbb{R}$, яка поточково прямує до нуля, існує така сукупно неперервна функція $f:[0,1]^{2}\rightarrow \mathbb{R}$, що $E_{n}(f^{x})=\alpha_{n}(x)$ на $[0,1]$ для кожного $ n$? Ця задача досі ще не розв'язана, отримано лише позитивний результат для скінченного числа функцій $\alpha_{0}, \alpha_{1},...,\alpha_{n}$.
\end{exampleblock}

\begin{exampleblock}{}
24. Волошин Г.А., Маслюченко В.К. Функціональне узагальнення однієї теореми Бернштейна // Мат. студії.~--- 2010.~--- \textbf{33},№2.~--- C. 220-224.
\end{exampleblock}
\end{frame}

\begin{frame}\frametitle{Формули для $E_{0}(f)$}
\begin{exampleblock}{}
Для довільної функції $f:X\rightarrow \mathbb{R}$, заданої на множині $X$ розглянемо числа $$M(f)=\sup\limits_{x\in X}^{} f(x)\textmd{, }m(f)=\inf\limits_{x\in X}^{} f(x)\textmd{ і } E_{0}(f)=\inf\{||f-c||:c\in \mathbb{R}\}.$$
\end{exampleblock}

\begin{exampleblock}{  \centerline{Теорема 16.}}
Нехай $X$~--- непорожня множина, $f:X\rightarrow \mathbb{R}$~--- обмежена функція і $\mu=\frac{1}{2}(M(f)-m(f))$. Тоді $E_{0}(f)=\mu$, при цьому стала функція $c_{0}=\frac{1}{2}(M(f)+m(f))$ є єдиною серед сталих функцій, для якої використовується рівність $$E_{0}(f)=\mu=||f-c_{0}||.$$
\end{exampleblock}

\end{frame}


\begin{frame}\frametitle{Пари Гана}
\begin{exampleblock}{}
Парою Гана на топологічному просторі $X$ ми називаємо таку пару $(g,h)$ функцій $g,h:X\rightarrow \mathbb{R}$, що $g(x)\leq h(x)$ на $X$, $g$~--- напівнеперервна зверху, $h$~--- напівнеперервна знизу. Будь-яка функція $f:X\rightarrow \mathbb{R}$, для якої $g(x)\leq f(x)\leq h(x)$ на $X$ називається проміжною для пари Гана $(g,h)$. Відомо, що $T_{1}$-простір $X$ є нормальним тоді і тільки тоді, коли кожна пара Гана $(g,h)$ на $X$ має неперервну проміжну функцію. Це твердження ми називаємо теоремою Гана-Д'єдонне-Тонґа-Катетова чи, коротше, теоремою Гана, який довів існування проміжної неперервної функції для кожної пари Гана у випадку метричного простору $X$.
\end{exampleblock}

\begin{exampleblock}{}
26. Hahn H. Uber halbstetige und unstetige Functionen // Sitzungsberichte Akad.Wiss.\\Wien.Math.~---naturwiss.Kl.Abt.IIa.~--- 1917.~--- \textbf{126}.~--- S.91-110.

27. Мельник В.С. Рівномірні відстані до різних функціональних класів та розвиток теореми Гана про проміжні функції// дис. канд. фіз.-мат. наук~--- Чернівці 2018. ~--- 111c. 
\end{exampleblock}
\end{frame}



\begin{frame}\frametitle{Пара Гана, що пов'язана з нарізно неперервною функцією}
\begin{exampleblock}{  \centerline{Теорема 16.}}
Нехай $X$~--- топологічний простір, $Y$~--- непорожня множина, $f:X\times Y\rightarrow \mathbb{R}$~--- неперервна за першою змінною і обмежена за другою змінною функція, а $g(x)=\inf\{f^{x}:y\in Y\}$ і $h(x)=\sup\{f^{x}(y):y\in Y\}$ на $X$. Тоді $(g,h)$~--- пара Гана на $X$.
\end{exampleblock}

\begin{exampleblock}{  \centerline{Наслідок.}}
Нехай $X$~--- топологічний простір, $Y$~--- непорожній компактний простір, $f:X\times Y\rightarrow \mathbb{R}$~--- нарізно неперервна функція і $g(x)=\min\limits_{y\in Y}^{}f^{x}(y)$, а $h(x)=\max\limits_{y\in Y}^{}f^{x}(y)$ на $X$. Тоді $(g,h)$~--- пара Гана на $X$.
\end{exampleblock}

\end{frame}


\begin{frame}\frametitle{Побідова нарізно неперервної функції з даними максимумами чи мінімумами}
\begin{exampleblock}{}
Для довільної функції $f:X\times Y\rightarrow \mathbb{R}$ визначимо функції $$S_{f}(x)=\sup\limits_{y\in Y}^{}f(x,y) та I_{f}(x)=\inf\limits_{y\in Y}^{}f(x,y).$$
Використовуючи один результат Тонґа про поточкову апроксимацію напівнеперервних функцій монотонними послідовностями неперервних функцій на досконало нормальному просторі $X$ і метод побудови нарізно неперервної функції з даною діагоналлю, що належить до першого класу Бера, ми отримуємо таку теорему.
\end{exampleblock}

\begin{exampleblock}{  \centerline{Теорема 17.}}
Нехай $h: X\rightarrow \mathbb{R}$ -- напівнеперервна знизу функція, що задана на досконало нормальному просторі $X$ з нормальним квадратом $X^2$, в якому діагональ $\Delta=\{(x,x): x\in X\}$ є $G_{\delta}$-множиною, і $h_0: X\rightarrow \mathbb{R}$ -- неперервна функція, для якої $h_0(x)\leq h(x)$ на $X$. Тоді існує така нарізно неперервна функція $f: X^2\rightarrow \mathbb{R}$, що $h_0(x)\leq f(x,y)\leq h(x)$ на $X^2$ i $S_f(x)=h(x)$ на $X$.
\end{exampleblock}
\end{frame}


\begin{frame}\frametitle{Побудова нарізно неперервної функції з даною парою Гана, що з нею пов'язана}
\begin{exampleblock}{  \centerline{Теорема 18.}}
Нехай $X=[a, b]$,  $Y=[c, d]$, де $a<b$, $c<d$ i $(g,h)$ -- довільна пара Гана на відрізку  $X$.  Тоді існує така нарізно неперервна функція $f: X\times Y\rightarrow\mathbb{R}$, що $(I_f, S_f)=(g, h)$.
\end{exampleblock}
\end{frame}



\begin{frame}\frametitle{Нарізно неперервні функції з даним нульовим наближенням}
\begin{exampleblock}{}
Для кожної нарізно неперервної функції $f: [a,b]\times [c,d]\rightarrow \mathbb{R}$ її нульове найкраще наближення $\alpha_f$ визначається на відрізку $X=[a,b]$ формулою
$$\alpha_f(x)=\frac{M(f^x)-m^(f^x)}{2}.$$
\end{exampleblock}

\begin{exampleblock}{  \centerline{Теорема 19.}}
Для довільного прямокутника $P=X\times Y$, де $X=[a, b]$ i $Y=[c, d]$ і кожної невід'ємної напівнеперервної знизу функції $\varphi: X \rightarrow \mathbb{R}$ існує така нарізно неперервна функція $f: P \rightarrow \mathbb{R}$, що $\alpha_f=\varphi$.
\end{exampleblock}
\end{frame}
