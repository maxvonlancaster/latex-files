\section*{Нерівності.}
\end{center}


\begin{enumerate}
\item Доведіть нервність $x^{4}+y^{4}+z^{4}\geq x^{2}yz+xy^{2}z+xyz^{2}$

\item Нехай $E$, $F$, $G$, $H$~--- середини сторін $AB$, $BC$, $CD$, $DA$ випуклого чотирикутника $ABCD$. Доведіть, що $S_{ABCD}\leq EG \cdot HF$.

\item Всередині трикутника $ABC$ взята точка $M$. Доведіть, що $4S\leq AM\cdot BC+ BM \cdot AC + CM\cdot AB$

\item Дано додатні числа $a$, $b$ та $c$ такі, що $abc=1$. Доведіть, що $$\frac{1}{1+a+b}+\frac{1}{1+b+c}+\frac{1}{1+c+a}\leq 1.$$

\item Дійсні числа $a,b,c,d$ по модулю більші за $1$ і задовільняють рівність $abc+abd+acd+bcd+a+b+c+d=0$. Доведіть нерівність $$\frac{1}{a-1}+\frac{1}{b-1}+\frac{1}{c-1}+\frac{1}{d-1}>0.$$



\end{enumerate}
