\textbf{Варіант I}


\begin{enumerate}
\item $\int\limits_{0}^{\frac{\pi}{2}}\cos 2x dx$

\item $\int\limits_{-\frac{1}{2}}^{\frac{1}{2}}\frac{dx}{\sqrt{1-x^{2}}}$


\item Заміна змінної: $\int\limits_{\ln 3}^{\ln 8}\frac{dx}{\sqrt{e^{x}+1}}$



\item  Інтегрування частинами: $\int\limits_{0}^{\frac{\pi}{2}} x \cos 2x dx$



\item Невласні інтеграли:
\begin{enumerate}
\item $\int\limits_{1}^{+\infty} \frac{dx}{\sqrt{x}}$

\item $\int\limits_{2}^{6} \frac{dx}{\sqrt{x-2}}$


\end{enumerate}
\end{enumerate}

\end{enumerate}
